2️⃣ Configurar os Workflows no AION

Os arquivos YAML que criamos (linkedin_workflow.yaml, outreach_workflow.yaml, content_workflow.yaml, site_integration.yaml) precisam ser carregados pelo orquestrador.

No código:
python

# src/orchestrator/workflow_runner.py
import yaml
from pathlib import Path

class WorkflowRunner:
    def __init__(self):
        self.workflows = self._load_workflows()
    
    def _load_workflows(self):
        workflows = {}
        for yaml_file in Path("config").glob("*_workflow.yaml"):
            with open(yaml_file) as f:
                workflows[yaml_file.stem] = yaml.safe_load(f)
        return workflows
    
    async def run(self, workflow_name: str):
        workflow = self.workflows.get(workflow_name)
        if not workflow:
            raise ValueError(f"Workflow {workflow_name} not found")
        
        for step in workflow.get("steps", []):
            agent = self._get_agent(step["agent"])
            result = await agent.execute(step["params"])

5️⃣ Ativar Agentes Após a Compra

Quando o cliente comprar, o webhook da Stripe ativa os agentes.

No webhook da Stripe:
python

# src/monetization/webhooks.py
@app.post("/webhook/stripe")
async def stripe_webhook(request: Request):
    payload = await request.body()
    sig_header = request.headers.get("stripe-signature")
    event = stripe.Webhook.construct_event(payload, sig_header, os.getenv("STRIPE_WEBHOOK_SECRET"))
    
    if event.type == "checkout.session.completed":
        session = event.data.object
        customer_email = session["customer_email"]
        plan = session["line_items"]["data"][0]["price"]["id"]
        
        # 1. Captura o lead
        lead = await save_lead_from_stripe(customer_email, plan)
        
        # 2. Ativa os agentes para o cliente
        await activate_agents_for_client(lead.id, plan)
        
        # 3. Notifica o time (Slack)
        await notify_slack(f"🎉 Nova venda! {customer_email} - {plan}")
    
    return {"status": "success"}
