# src/agents/optimization/ab_analyzer.py
"""
Agente de Otimização Contínua — Lê testes A/B, identifica gargalos e aplica correções
"""

import pandas as pd
import requests
from typing import Dict, Any, List
from datetime import datetime, timedelta
import yaml

class ABTestAnalyzer:
    """
    Agente que analisa testes A/B e otimiza o site automaticamente
    """
    
    def __init__(self):
        self.metrics = {
            "bounce_rate": {"threshold": 0.60, "action": "increase_engagement"},
            "conversion_rate": {"threshold": 0.02, "action": "simplify_checkout"},
            "click_through_rate": {"threshold": 0.05, "action": "improve_cta"},
            "abandonment_rate": {"threshold": 0.70, "action": "reduce_friction"},
            "time_on_page": {"threshold": 45, "action": "improve_content"},
            "form_completion": {"threshold": 0.30, "action": "simplify_form"}
        }
        self.data_sources = {
            "ga4": self._fetch_ga4_data,
            "supabase": self._fetch_supabase_data,
            "stripe": self._fetch_stripe_data
        }
    
    def run_analysis(self) -> Dict[str, Any]:
        """
        Executa análise completa dos testes A/B
        """
        # 1. Coleta dados de todas as fontes
        data = self._collect_all_data()
        
        # 2. Analisa cada variante
        variants = self._analyze_variants(data)
        
        # 3. Identifica gargalos
        bottlenecks = self._identify_bottlenecks(variants)
        
        # 4. Gera recomendações
        recommendations = self._generate_recommendations(bottlenecks)
        
        # 5. Aplica correções
        changes = self._apply_fixes(recommendations)
        
        return {
            "variants": variants,
            "bottlenecks": bottlenecks,
            "recommendations": recommendations,
            "changes": changes,
            "deploy_requested": True
        }
    
    def _analyze_variants(self, data: Dict) -> Dict:
        """
        Analisa cada variante do teste A/B
        """
        variants = {}
        
        for variant_id, variant_data in data.items():
            score = self._calculate_variant_score(variant_data)
            
            variants[variant_id] = {
                "score": score,
                "conversion_rate": variant_data.get("conversion_rate", 0),
                "bounce_rate": variant_data.get("bounce_rate", 0),
                "avg_time": variant_data.get("avg_time", 0),
                "ctr": variant_data.get("ctr", 0),
                "issues": self._find_issues(variant_data)
            }
        
        return variants
    
    def _identify_bottlenecks(self, variants: Dict) -> List[Dict]:
        """
        Identifica gargalos em cada variante
        """
        bottlenecks = []
        winning_variant = max(variants.items(), key=lambda x: x[1]["score"])
        
        for variant_id, data in variants.items():
            if variant_id == winning_variant[0]:
                continue
            
            issues = data.get("issues", [])
            for issue in issues:
                bottlenecks.append({
                    "variant": variant_id,
                    "issue": issue["type"],
                    "severity": issue["severity"],
                    "impact": issue["impact"],
                    "vs_winner": data["score"] - winning_variant[1]["score"]
                })
        
        return sorted(bottlenecks, key=lambda x: x["impact"], reverse=True)
    
    def _generate_recommendations(self, bottlenecks: List[Dict]) -> List[Dict]:
        """
        Gera recomendações baseadas nos gargalos identificados
        """
        recommendations = []
        
        for bottleneck in bottlenecks[:5]:  # Top 5 gargalos
            issue_type = bottleneck["issue"]
            
            if issue_type == "high_bounce":
                recommendations.append({
                    "action": "add_trust_signals",
                    "description": "Adicionar selos de confiança, depoimentos e cases",
                    "file": "index.html",
                    "section": "hero"
                })
            elif issue_type == "low_conversion":
                recommendations.append({
                    "action": "simplify_checkout",
                    "description": "Reduzir campos do checkout e adicionar botão de compra direta",
                    "file": "index.html",
                    "section": "pricing"
                })
            elif issue_type == "low_ctr":
                recommendations.append({
                    "action": "improve_cta",
                    "description": "Alterar texto do CTA para ser mais direto e urgente",
                    "file": "index.html",
                    "section": "cta"
                })
            elif issue_type == "high_abandonment":
                recommendations.append({
                    "action": "add_progress_indicator",
                    "description": "Adicionar indicador de progresso no checkout",
                    "file": "checkout.html",
                    "section": "checkout"
                })
        
        return recommendations
    
    def _apply_fixes(self, recommendations: List[Dict]) -> List[Dict]:
        """
        Aplica as correções no site
        """
        changes = []
        
        for rec in recommendations:
            file_path = f"site/{rec['file']}"
            
            # Lê o arquivo atual
            with open(file_path, 'r') as f:
                content = f.read()
            
            # Aplica a correção
            modified_content = self._apply_change(content, rec)
            
            # Salva o arquivo modificado
            with open(file_path, 'w') as f:
                f.write(modified_content)
            
            changes.append({
                "file": rec["file"],
                "action": rec["action"],
                "description": rec["description"],
                "applied": True
            })
        
        return changes
    
    def _apply_change(self, content: str, recommendation: Dict) -> str:
        """
        Aplica uma mudança específica no código
        """
        action = recommendation["action"]
        
        if action == "add_trust_signals":
            # Adiciona selos de confiança após o hero
            trust_section = """
            <div class="trust-signals" style="display:flex;justify-content:center;gap:40px;padding:20px;background:#f8fafc;border-radius:12px;margin:20px 0">
                <div><span style="font-weight:700;">500+</span> clientes</div>
                <div><span style="font-weight:700;">99.9%</span> uptime</div>
                <div><span style="font-weight:700;">24/7</span> suporte</div>
                <div><span style="font-weight:700;">7 dias</span> grátis</div>
            </div>
            """
            # Insere após a seção hero
            content = content.replace('</section>', trust_section + '</section>', 1)
            
        elif action == "improve_cta":
            # Altera texto do CTA
            content = content.replace('Começar Agora', 'Teste Grátis por 7 Dias 🚀')
            content = content.replace('Ver Funcionalidades', 'Ver Planos e Preços')
            
        elif action == "simplify_checkout":
            # Adiciona botão de compra direta
            buy_button = """
            <button onclick="window.location.href='/checkout'" 
                    style="background:#2563eb;color:#fff;padding:16px 48px;border:none;border-radius:50px;font-size:20px;font-weight:700;cursor:pointer">
                Comprar Agora → R$ 2.391/mês

            """
            content = content.replace('</section>', buy_button + '</section>', 1)
        
        return content

# src/orchestrator/optimization_runner.py
"""
Orquestrador que roda o agente de otimização continuamente
"""

import asyncio
import schedule
from src.agents.optimization.ab_analyzer import ABTestAnalyzer
from src.monetization.stripe_client import StripeClient
from src.database.supabase_client import supabase

class OptimizationOrchestrator:
    """
    Orquestra a análise contínua e otimização do site
    """
    
    def __init__(self):
        self.analyzer = ABTestAnalyzer()
        self.stripe = StripeClient()
        self.last_run = None
        self.deploy_queue = []
    
    async def run_daily_optimization(self):
        """
        Roda a otimização diariamente
        """
        print("🔍 Iniciando análise de otimização...")
        
        # 1. Coleta dados
        data = self._collect_metrics()
        
        # 2. Analisa
        result = self.analyzer.run_analysis()
        
        # 3. Se detectou gargalos, aplica correções
        if result["bottlenecks"]:
            print(f"⚠️ {len(result['bottlenecks'])} gargalos identificados")
            
            # Aplica correções
            changes = self.analyzer._apply_fixes(result["recommendations"])
            
            # Prepara deploy
            self._request_deploy(changes)
            
            # Notifica
            await self._notify_slack(result)
        
        self.last_run = datetime.now()
        return result
    
    def _collect_metrics(self) -> Dict:
        """
        Coleta métricas de todas as fontes
        """
        return {
            "ga4": self._get_ga4_metrics(),
            "supabase": self._get_supabase_metrics(),
            "stripe": self._get_stripe_metrics()
        }
    
    def _request_deploy(self, changes: List[Dict]):
        """
        Solicita deploy para Netlify com as mudanças
        """
        print("🚀 Solicitando deploy para Netlify...")
        
        # Gera commit message
        commit_msg = "🤖 Otimização automática: " + ", ".join([c["action"] for c in changes])
        
        # Prepara arquivo de deploy
        deploy_config = {
            "changes": changes,
            "commit": commit_msg,
            "timestamp": datetime.now().isoformat()
        }
        
        # Salva configuração
        with open("deploy_request.json", "w") as f:
            import json
            json.dump(deploy_config, f, indent=2)
        
        # Executa deploy
        import subprocess
        subprocess.run(["bash", "scripts/deploy_netlify.sh"])
        
        print("✅ Deploy solicitado com sucesso!")
    
    async def _notify_slack(self, result: Dict):
        """
        Notifica via Slack sobre as otimizações
        """
        message = f"""
        🤖 *Otimização Automática AION*
        
        📊 *Gargalos Identificados:* {len(result['bottlenecks'])}
        🔧 *Correções Aplicadas:* {len(result['changes'])}
        🚀 *Deploy:* Solicitado
        
        *Detalhes:*
        """
        for change in result["changes"][:3]:
            message += f"\n- {change['action']}: {change['description']}"
        
        # Envia para Slack
        # webhook = os.getenv("SLACK_WEBHOOK_URL")
        # requests.post(webhook, json={"text": message})
        
        print(message)

# scripts/run_optimization.py
"""
Script para rodar a otimização automaticamente
"""

import asyncio
import schedule
import time
from src.orchestrator.optimization_runner import OptimizationOrchestrator

async def main():
    orchestrator = OptimizationOrchestrator()
    
    # Roda a cada 24 horas
    schedule.every().day.at("06:00").do(
        lambda: asyncio.create_task(orchestrator.run_daily_optimization())
    )
    
    # Também roda manualmente se chamado com --force
    import sys
    if "--force" in sys.argv:
        await orchestrator.run_daily_optimization()
    
    # Mantém rodando
    while True:
        schedule.run_pending()
        time.sleep(60)

if __name__ == "__main__":
    asyncio.run(main())
