# src/orchestrator/graph.py
from langgraph.graph import StateGraph, END
from typing import TypedDict, List, Dict, Any
import asyncio

class AgentState(TypedDict):
    query: str
    plan: List[Dict[str, Any]]
    documents: List[Dict[str, Any]]
    synthesis: str
    action_results: Dict[str, Any]
    metadata: Dict[str, Any]
    quality_status: bool
    quality_report: str

def create_multi_agent_graph():
    workflow = StateGraph(AgentState)
    
    workflow.add_node("planner", planner_node)
    workflow.add_node("router", router_node)
    workflow.add_node("coordinator", coordinator_node)
    workflow.add_node("quality_check", quality_check_node)
    workflow.add_node("synthesizer", synthesizer_node)
    
    workflow.set_entry_point("planner")
    workflow.add_edge("planner", "router")
    workflow.add_edge("router", "coordinator")
    workflow.add_edge("coordinator", "quality_check")
    workflow.add_edge("quality_check", "synthesizer")
    workflow.add_edge("synthesizer", END)
    
    return workflow.compile()

# src/security/aip_registry.py
from aip_identity import AIPClient
import uuid

class AIPRegistry:
    def __init__(self):
        self.registry_url = "https://aip.aion.global"
        self.agents = {}
    
    def register_agent(self, agent_id: str, agent_name: str, principal: str) -> dict:
        client = AIPClient.register(self.registry_url, agent_name)
        client.save(f"credentials/{agent_id}.json")
        
        self.agents[agent_id] = {
            "did": client.did,
            "public_key": client.public_key,
            "principal": principal,
            "status": "active"
        }
        return self.agents[agent_id]
    
    def verify_agent(self, agent_id: str, signature: bytes) -> bool:
        if agent_id not in self.agents:
            return False
        return self.agents[agent_id]["status"] == "active"
    
    def revoke_agent(self, agent_id: str):
        if agent_id in self.agents:
            self.agents[agent_id]["status"] = "revoked"

# src/quality/orchestrator.py
class QualityOrchestrator:
    def __init__(self):
        self.gates = self._load_gates()
        self.critic = QualityCritic()
    
    async def execute(self, agent_output: Dict[str, Any], agent_id: str) -> Dict[str, Any]:
        # 1. Apply gates
        gate_results = await self._apply_gates(agent_output, agent_id)
        
        # 2. If failed, attempt auto-fix
        if not gate_results["passed"]:
            corrected = await self._attempt_fix(agent_output, gate_results)
            gate_results = await self._apply_gates(corrected, agent_id)
            agent_output = corrected
        
        # 3. Critic review
        critic_review = await self.critic.review(agent_output, agent_id)
        
        return {
            "output": agent_output,
            "quality_status": gate_results["passed"] and critic_review["passed"],
            "gate_report": gate_results["report"],
            "critic_report": critic_review["report"]
        }
    
    def _load_gates(self) -> dict:
        return {
            "coverage": {"min": 80},
            "security_issues": {"max": 0},
            "duplication": {"max": 3},
            "reliability_issues": {"max": 0}
        }

from supabase import create_client
import os

class SupabaseClient:
    def __init__(self):
        self.url = os.getenv("SUPABASE_URL")
        self.key = os.getenv("SUPABASE_KEY")
        self.client = create_client(self.url, self.key)
    
    def get_subscriptions(self, status: str = "active"):
        return self.client.table("subscriptions")\
            .select("*")\
            .eq("status", status)\
            .execute()
    
    def create_subscription(self, data: dict):
        return self.client.table("subscriptions").insert(data).execute()

9. COMANDOS DE IMPLEMENTAÇÃO
9.1 Instalação
bash

# Clone e instale
git clone https://github.com/global-engenharia/aion.git
cd aion

# Criar ambiente virtual
python -m venv venv
source venv/bin/activate  # Windows: venv\Scripts\activate

# Instalar dependências
pip install -r requirements.txt

# Configurar variáveis de ambiente
cp .env.example .env
# Editar .env com suas credenciais

9.2 Inicializar Agentes
bash

# Inicializar todos os agentes
python scripts/init_agents.py

# Inicializar grupo específico
python scripts/init_agents.py --group aec
python scripts/init_agents.py --group regulatory
python scripts/init_agents.py --group microsoft

9.3 Rodar Orquestrador
bash

# Rodar localmente
python src/main.py --config config.yaml

# Modo debug
python src/main.py --config config.yaml --debug

# Testar agente específico
python src/main.py --agent #1 --input "analisar documento"

bash

pytest tests/test_agents.py -v

bash

# Docker build
