"""
AIP Layer 2: Enforcement Proxy
"""

from typing import Dict, Any, Optional
from datetime import datetime
from enum import Enum

class Decision(Enum):
    ALLOW = "allow"
    DENY = "deny"
    ASK = "ask"

class AIPProxy:
    def __init__(self, registry: AIPRegistry):
        self.registry = registry
        self.audit_log = []
    
    async def intercept_tool_call(self, tool_call: Dict[str, Any], aat_token: str) -> Decision:
        # 1. Verifica AAT
        if not self.registry.verify_token(aat_token):
            return Decision.DENY
        
        # 2. Extrai claims
        claims = self._decode_token(aat_token)
        
        # 3. Verifica budget
        if claims.get("budget_usd", 0) <= 0:
            return Decision.DENY
        
        # 4. Verifica capabilities
        if tool_call.get("tool") not in claims.get("capabilities", []):
            return Decision.DENY
        
        # 5. Registra
        self._audit(claims["agent_id"], tool_call, Decision.ALLOW)
        
        return Decision.ALLOW
    
    def _audit(self, agent_id: str, tool_call: Dict[str, Any], decision: Decision):
        self.audit_log.append({
            "agent_id": agent_id,
            "tool": tool_call.get("tool"),
            "decision": decision.value,
            "timestamp": datetime.now().isoformat()
        })

# src/security/aip/token.py
"""
AAT (Agent Authentication Token) — Generation and Validation
"""

import jwt
import os
from datetime import datetime, timedelta
from typing import Dict, Any

class AATToken:
    SECRET_KEY = os.getenv("AIP_SECRET_KEY", "default-secret-change-me")
    
    @classmethod
    def generate(cls, agent_id: str, capabilities: list, budget_usd: float, expiry_seconds: int = 3600) -> str:
        payload = {
            "agent_id": agent_id,
            "capabilities": capabilities,
            "budget_usd": budget_usd,
            "exp": datetime.now() + timedelta(seconds=expiry_seconds),
            "iat": datetime.now()
        }
        return jwt.encode(payload, cls.SECRET_KEY, algorithm="HS256")
    
    @classmethod
    def decode(cls, token: str) -> Dict[str, Any]:
        return jwt.decode(token, cls.SECRET_KEY, algorithms=["HS256"])

# src/security/aip/scope.py
"""
Scope Attenuation — Capabilities só podem diminuir
"""

from typing import List, Set

class ScopeAttenuation:
    @staticmethod
    def validate_chain(parent_capabilities: List[str], child_capabilities: List[str]) -> bool:
        return set(child_capabilities).issubset(set(parent_capabilities))
    
    @staticmethod
    def attenuate(parent_tools: List[str], child_tools: List[str]) -> List[str]:
        if not ScopeAttenuation.validate_chain(parent_tools, child_tools):
            raise ValueError("Child cannot have capabilities outside parent scope")
        return child_tools

"""
Compliance Receipts — IETF draft-marques-asqav-compliance-receipts-05
Campos OBRIGATÓRIOS para auditoria multi-jurisdição
"""

from datetime import datetime
from typing import Dict, Any, Optional
import hashlib

class ComplianceReceipt:
    def __init__(self):
        self.chain = []
        self.merkle_root = None
    
    def create_receipt(self, action: Dict[str, Any], decision: str, agent_id: str, risk_classification: str = "low") -> Dict[str, Any]:
        receipt = {
            # REQUIRED
            "type": "compliance",
            "issued_at": datetime.now().isoformat(),
            "issuer_id": agent_id,
            
            # OPTIONAL → REQUIRED no profile
            "action": {
                "tool": action.get("tool"),
                "arguments": action.get("arguments"),
                "outcome": action.get("outcome")
            },
            "decision": decision,
            
            # Timestamp anchors (RFC 3161)
            "timestamp_anchor": datetime.now().isoformat(),
            
            # Extensions
            "risk_classification": risk_classification,
            "previous_receipt_hash": self._get_previous_hash(),
            "build_provenance": os.getenv("BUILD_VERSION", "unknown")
        }
        
        # Calcula hash
        receipt["hash"] = self._calculate_hash(receipt)
        self.chain.append(receipt)
        return receipt
    
    def _get_previous_hash(self) -> str:
        return self.chain[-1]["hash"] if self.chain else "0000000000000000"
    
    def _calculate_hash(self, receipt: Dict[str, Any]) -> str:
        return hashlib.sha256(json.dumps(receipt, sort_keys=True).encode()).hexdigest()
    
    def verify_chain(self) -> bool:
        for i in range(1, len(self.chain)):
            if self.chain[i]["previous_receipt_hash"] != self.chain[i-1]["hash"]:
                return False
        return True

from datetime import datetime
from typing import Dict, Any, List

class AgentInventory:
    def __init__(self):
        self.agents: Dict[str, Dict[str, Any]] = {}
    
    def register_agent(self, agent_id: str, name: str, capabilities: List[str], principal: str, group: str = "default"):
        self.agents[agent_id] = {
            "agent_id": agent_id,
            "name": name,
            "capabilities": capabilities,
            "principal": principal,
            "group": group,
            "status": "active",
            "created_at": datetime.now().isoformat(),
            "last_active": datetime.now().isoformat()
        }
        return self.agents[agent_id]
    
    def revoke_agent(self, agent_id: str):
        if agent_id in self.agents:
            self.agents[agent_id]["status"] = "revoked"
            self.agents[agent_id]["revoked_at"] = datetime.now().isoformat()
    
    def get_report(self) -> Dict[str, Any]:
        active = [a for a in self.agents.values() if a["status"] == "active"]
        return {
            "total": len(self.agents),
            "active": len(active),
            "revoked": len([a for a in self.agents.values() if a["status"] == "revoked"]),
            "by_group": self._group_by("group"),
            "by_principal": self._group_by("principal"),
            "report_timestamp": datetime.now().isoformat()
        }
    
    def _group_by(self, key: str) -> Dict[str, int]:
        groups = {}
        for agent in self.agents.values():
            group_key = agent.get(key, "unknown")
            groups[group_key] = groups.get(group_key, 0) + 1
        return groups

# src/security/rag/sanitization.py
"""
Tratar RAG como Hostil — IETF Best Practices
Todo conteúdo externo é tratado como hostil até prova contrária
"""

from typing import List, Dict, Any
import re

class RAGSanitizer:
    def __init__(self):
        self.suspicious_patterns = [
            r"ignore previous instructions",
            r"new instructions",
            r"system prompt",
            r"override",
            r"you are now",
            r"forget all previous"
        ]
    
    def validate_source(self, source: Dict[str, Any]) -> bool:
        """Valida se a fonte é confiável"""
        required_fields = ["url", "timestamp", "hash"]
        for field in required_fields:
            if field not in source:
                return False
        return True
    
    def sanitize_chunk(self, chunk: str) -> str:
        """Remove padrões suspeitos do chunk"""
        for pattern in self.suspicious_patterns:
            chunk = re.sub(pattern, "[REDACTED]", chunk, flags=re.IGNORECASE)
        return chunk
    
    def process_document(self, content: str, source: Dict[str, Any]) -> Dict[str, Any]:
        """Processa documento com validação de fonte"""
        if not self.validate_source(source):
            return {"error": "Invalid source", "sanitized": False}
        
        chunks = self._chunk_document(content)
        sanitized_chunks = [self.sanitize_chunk(chunk) for chunk in chunks]
        
        return {
            "source": source,
            "chunks": sanitized_chunks,
            "sanitized": True,
            "provenance": self._create_provenance(source)
        }
    
    def _chunk_document(self, content: str, chunk_size: int = 1000) -> List[str]:
        return [content[i:i+chunk_size] for i in range(0, len(content), chunk_size)]
    
    def _create_provenance(self, source: Dict[str, Any]) -> Dict[str, Any]:
        return {
            "source_url": source.get("url"),
            "source_hash": source.get("hash"),
            "retrieved_at": source.get("timestamp"),
            "sanitized_at": datetime.now().isoformat()
        }

# src/security/output/validator.py
"""
Output Validator — Valida antes de qualquer ação com efeito colateral
"""

from typing import Dict, Any, List
from enum import Enum

class ValidationResult(Enum):
    APPROVED = "approved"
    REJECTED = "rejected"
    REVIEW_NEEDED = "review_needed"

class OutputValidator:
    def __init__(self):
        self.high_impact_actions = ["delete", "write", "update", "create", "execute"]
        self.sensitive_patterns = [
            r"\b\d{3}\.\d{3}\.\d{3}-\d{2}\b",  # CPF
            r"\b\d{2}\.\d{3}\.\d{3}/\d{4}-\d{2}\b",  # CNPJ
            r"\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b"  # Email
        ]
    
    def validate_output(self, output: Dict[str, Any], action: Dict[str, Any]) -> ValidationResult:
        # 1. Verifica se é ação de alto impacto
        if self._is_high_impact(action):
            # 2. Verifica dados sensíveis
            if self._contains_sensitive_data(output):
                return ValidationResult.REVIEW_NEEDED
            
            # 3. Verifica cobertura de testes
            if not self._has_test_coverage(output):
                return ValidationResult.REVIEW_NEEDED
            
            # 4. Verifica schema
            if not self._validate_schema(output):
                return ValidationResult.REJECTED
        
        return ValidationResult.APPROVED
    
    def _is_high_impact(self, action: Dict[str, Any]) -> bool:
        action_type = action.get("type", "").lower()
        return any(pattern in action_type for pattern in self.high_impact_actions)
    
    def _contains_sensitive_data(self, output: Dict[str, Any]) -> bool:
        output_str = json.dumps(output)
        for pattern in self.sensitive_patterns:
            if re.search(pattern, output_str):
                return True
        return False
    
    def _has_test_coverage(self, output: Dict[str, Any]) -> bool:
        # Verifica se há testes para o output
        return output.get("tested", False)
    
    def _validate_schema(self, output: Dict[str, Any]) -> bool:
        # Verifica se o output segue o schema esperado
        required_fields = ["status", "result"]
        return all(field in output for field in required_fields)

# src/security/router/guardrail.py
"""
Guardrail Router — NIST CAISI
Classifica requisições: Lookup vs Reasoning
"""

from typing import Dict, Any
from enum import Enum

class RequestType(Enum):
    LOOKUP = "lookup"
    REASONING = "reasoning"
    BLOCKED = "blocked"

class GuardrailRouter:
    def __init__(self):
        self.lookup_keywords = ["consulta", "buscar", "obter", "verificar", "mostrar"]
        self.reasoning_keywords = ["analisar", "raciocinar", "planejar", "criar", "desenvolver"]
        self.blocked_patterns = ["delete all", "drop table", "rm -rf"]
    
    def route(self, request: Dict[str, Any]) -> RequestType:
        content = request.get("content", "").lower()
        source = request.get("source", "unknown")
        
        # 1. Verifica bloqueios
        for pattern in self.blocked_patterns:
            if pattern in content:
                return RequestType.BLOCKED
        
        # 2. Classifica por palavras-chave
        for keyword in self.lookup_keywords:
            if keyword in content:
                return RequestType.LOOKUP
        
        for keyword in self.reasoning_keywords:
            if keyword in content:
                return RequestType.REASONING
        
        # 3. Por padrão, usa reasoning
        return RequestType.REASONING
    
    def get_constraints(self, request_type: RequestType) -> Dict[str, Any]:
        if request_type == RequestType.LOOKUP:
            return {
                "max_tokens": 1000,
                "no_tools": True,
                "budget_usd": 0.001
            }
        elif request_type == RequestType.REASONING:
            return {
                "max_tokens": 5000,
                "tools": ["search", "compute"],
                "budget_usd": 0.01
            }
        else:
            return {
                "blocked": True,
                "reason": "policy_violation"
            }

# src/security/communication/mtls.py
"""
mTLS entre Agentes — IETF AIP
"""

import ssl
import aiohttp
from typing import Dict, Any
import os

class MTLSServer:
    def __init__(self):
        self.ssl_context = self._create_ssl_context()
    
    def _create_ssl_context(self) -> ssl.SSLContext:
        ctx = ssl.create_default_context(ssl.Purpose.CLIENT_AUTH)
        ctx.load_cert_chain(
            os.getenv("MTLS_CERT_PATH", "certs/server.crt"),
            os.getenv("MTLS_KEY_PATH", "certs/server.key")
        )
        ctx.load_verify_locations(
            os.getenv("MTLS_CA_PATH", "certs/ca.crt")
        )
        ctx.verify_mode = ssl.CERT_REQUIRED
        return ctx
    
    async def secure_connect(self, url: str) -> aiohttp.ClientResponse:
        connector = aiohttp.TCPConnector(ssl=self.ssl_context)
        async with aiohttp.ClientSession(connector=connector) as session:
            return await session.get(url)

class MTLSClient:
    def __init__(self):
        self.ssl_context = self._create_ssl_context()
    
    def _create_ssl_context(self) -> ssl.SSLContext:
        ctx = ssl.create_default_context(ssl.Purpose.SERVER_AUTH)
        ctx.load_cert_chain(
            os.getenv("MTLS_CERT_PATH", "certs/client.crt"),
            os.getenv("MTLS_KEY_PATH", "certs/client.key")
        )
        ctx.load_verify_locations(
            os.getenv("MTLS_CA_PATH", "certs/ca.crt")
        )
        ctx.verify_mode = ssl.CERT_REQUIRED
        return ctx

    workflow.add_edge("planner", "executor")
    workflow.add_edge("executor", "validator")
    workflow.add_edge("validator", "compliance")
    workflow.add_edge("compliance", END)
    
    return workflow.compile()
